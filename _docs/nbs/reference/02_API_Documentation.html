API Documentation

API Documentation

This document provides comprehensive API documentation for the ML Deploy platform, including contracts, interfaces, and implementation details for all major components.

Overview

The ML Deploy platform uses a contract-based architecture with clear interfaces between components. All APIs follow RESTful conventions where appropriate, with Python dataclasses for type safety and serialization.

Core APIs

Web UI Contracts

NotebookExecutionRequest

from dataclasses import dataclass
from typing import Optional, Dict, Any, List
from pathlib import Path
from datetime import datetime

@dataclass(frozen=True)
class NotebookExecutionRequest:
    """Request for notebook execution with external parameter injection."""
    
    # Notebook reference
    notebook_ref: str  # Git commit SHA or approved reference
    notebook_path: Path  # Path to notebook within repository
    
    # Execution parameters
    parameters: Dict[str, Any]  # External parameter injection
    environment: str  # dev/uat/regression/prod
    
    # Resource requirements
    resource_requirements: Dict[str, Any]  # CPU, memory, GPU requirements
    
    # Metadata
    user_id: str
    request_id: str
    timestamp: datetime
    
    # Optional configuration
    execution_timeout: Optional[int] = None  # seconds
    priority: Optional[str] = "normal"  # low/normal/high/critical
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for serialization."""
        return {
            "notebook_ref": self.notebook_ref,
            "notebook_path": str(self.notebook_path),
            "parameters": self.parameters,
            "environment": self.environment,
            "resource_requirements": self.resource_requirements,
            "user_id": self.user_id,
            "request_id": self.request_id,
            "timestamp": self.timestamp.isoformat(),
            "execution_timeout": self.execution_timeout,
            "priority": self.priority
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "NotebookExecutionRequest":
        """Create from dictionary."""
        return cls(
            notebook_ref=data["notebook_ref"],
            notebook_path=Path(data["notebook_path"]),
            parameters=data["parameters"],
            environment=data["environment"],
            resource_requirements=data["resource_requirements"],
            user_id=data["user_id"],
            request_id=data["request_id"],
            timestamp=datetime.fromisoformat(data["timestamp"]),
            execution_timeout=data.get("execution_timeout"),
            priority=data.get("priority", "normal")
        )

ExecutionOrchestrator

from dataclasses import dataclass
from typing import Dict, Any, List, Optional
from enum import Enum

class ExecutionStatus(Enum):
    PENDING = "pending"
    SUBMITTED = "submitted"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"
    TIMEOUT = "timeout"

@dataclass
class ExecutionOrchestrator:
    """Manages notebook execution across different backends."""
    
    def __init__(self):
        self.active_runs: Dict[str, Dict[str, Any]] = {}
    
    def submit_execution(self, request: NotebookExecutionRequest) -> Dict[str, Any]:
        """Submit a notebook execution request."""
        run_id = f"run_{request.request_id}"
        
        # Determine backend based on environment and requirements
        backend = self._select_backend(request)
        
        # Submit to backend
        submission_result = backend.submit(request)
        
        # Track run
        self.active_runs[run_id] = {
            "request": request,
            "backend": backend,
            "status": ExecutionStatus.PENDING,
            "start_time": datetime.now(),
            "end_time": None,
            "result": None
        }
        
        return {
            "run_id": run_id,
            "status": ExecutionStatus.PENDING.value,
            "backend": backend.name,
            "submission_time": datetime.now().isoformat()
        }
    
    def get_run_status(self, run_id: str) -> Dict[str, Any]:
        """Get the status of a specific run."""
        if run_id not in self.active_runs:
            raise ValueError(f"Run {run_id} not found")
        
        run = self.active_runs[run_id]
        return {
            "run_id": run_id,
            "status": run["status"].value,
            "start_time": run["start_time"].isoformat(),
            "end_time": run["end_time"].isoformat() if run["end_time"] else None,
            "backend": run["backend"].name
        }
    
    def get_run_result(self, run_id: str) -> Dict[str, Any]:
        """Get the result of a completed run."""
        if run_id not in self.active_runs:
            raise ValueError(f"Run {run_id} not found")
        
        run = self.active_runs[run_id]
        if run["status"] not in [ExecutionStatus.COMPLETED, ExecutionStatus.FAILED]:
            raise ValueError(f"Run {run_id} not completed")
        
        return run["result"]
    
    def cancel_run(self, run_id: str) -> bool:
        """Cancel a running run."""
        if run_id not in self.active_runs:
            return False
        
        run = self.active_runs[run_id]
        if run["status"] in [ExecutionStatus.COMPLETED, ExecutionStatus.FAILED]:
            return False
        
        # Cancel in backend
        run["backend"].cancel(run_id)
        run["status"] = ExecutionStatus.CANCELLED
        run["end_time"] = datetime.now()
        
        return True
    
    def _select_backend(self, request: NotebookExecutionRequest) -> Any:

Governance Gates

RepositoryGovernanceState

from dataclasses import dataclass
from typing import Literal

RepositoryPhase = Literal["specification_first", "implementation_enabled"]
OperationKind = Literal["implementation_blocked", "implementation_allowed"]

@dataclass(frozen=True)
class RepositoryGovernanceState:
    """Current state of the repository governance system."""
    
    phase: RepositoryPhase = "specification_first"
    spec_quality_gate_passed: bool = False
    
    def can_implementation_proceed(self, explicit_user_confirmation: bool) -> bool:
        """Check if implementation can proceed."""
        if self.phase == "implementation_enabled":
            return True
        
        if not explicit_user_confirmation:
            return False
        
        return self.spec_quality_gate_passed

GovernanceDecision

from dataclasses import dataclass
from typing import Literal

@dataclass(frozen=True)
class GovernanceDecision:
    """Decision result from governance gate evaluation."""
    
    allowed: bool
    operation: OperationKind
    reason: str
    next_state: RepositoryGovernanceState

def request_implementation_transition(
    state: RepositoryGovernanceState,
    *,
    explicit_user_confirmation: bool,
) -> GovernanceDecision:
    """Enforce spec-first phase transition gate."""
    if state.phase == "implementation_enabled":
        return GovernanceDecision(
            allowed=True,
            operation="implementation_allowed",
            reason="already implementation_enabled",
            next_state=state,
        )

    if not explicit_user_confirmation:
        return GovernanceDecision(
            allowed=False,
            operation="implementation_blocked",
            reason="explicit_spec_confirmation_required",
            next_state=state,
        )

    if not state.spec_quality_gate_passed:
        return GovernanceDecision(
            allowed=False,
            operation="implementation_blocked",
            reason="spec_quality_gate_not_passed",
            next_state=state,
        )

    return GovernanceDecision(
        allowed=True,
        operation="implementation_allowed",
        reason="explicit_confirmation_with_quality_gate",
        next_state=RepositoryGovernanceState(
            phase="implementation_enabled",
            spec_quality_gate_passed=state.spec_quality_gate_passed,
        ),
    )

            "error_message": str(error),
            "context": context or {},
            "timestamp": datetime.now().isoformat(),
            "stack_trace": ErrorHandler._get_stack_trace()
        }
    
    @staticmethod
    def _get_stack_trace() -> str:
        """Get stack trace for error."""
        import traceback
        return traceback.format_exc()
    
    @staticmethod
    def log_error(error: Exception, context: Dict[str, Any] = None):
        """Log an error with context."""
        formatted_error = ErrorHandler.format_error(error, context)
        logger = MLLogger(Path("/tmp/logs"))
        logger.log_error("operation_error", error, formatted_error)

from typing import Dict, Any, List
from pydantic import BaseModel, validator

class ValidatedConfig(BaseModel):
    """Base class for validated configurations."""
    
    @classmethod
    def validate_config(cls, config: Dict[str, Any]) -> 'ValidatedConfig':
        """Validate configuration dictionary."""
        try:
            return cls(**config)
        except ValidationError as e:
            raise ConfigurationError(f"Configuration validation failed: {e}")

class MLflowConfig(ValidatedConfig):
    """Validated MLflow configuration."""
    tracking_uri: str
    artifact_root: str
    backend: str
    database_url: Optional[str] = None
    
    @validator('backend')
    def validate_backend(cls, v):
        if v not in ['sqlite', 'postgresql']:
            raise ValueError('Backend must be sqlite or postgresql')
        return v
